MySQL 用户基本操作笔记

# MySQL 用户基本操作笔记

# 实例-MySQL 用户基本操作

# 1、MySQL创建数据库与创建用户以及授权语句汇总

--1、创建数据库
create schema [数据库名称] default character set utf8 collate utf8_general_ci;
--创建数据库采用create schema和create database创建数据库的效果一样。

--2、创建用户
create user '[用户名称]'@'%' identified by '[用户密码]';
--创建用户
--密码8位以上，包括：大写字母、小写字母、数字、特殊字符
--%：匹配所有主机，该地方还可以设置成‘localhost’，代表只能本地访问，例如root账户默认为‘localhost‘

<!-- more -->

--3、用户授权
grant select,insert,update,delete,create on [数据库名称].* to [用户名称];
--用户授权数据库
--*代表整个数据库

--4、刷新权限
flush  privileges;
--立即启用修改

--5、取消用户权限
revoke all on *.* from tester;
--取消用户权限所有数据库（表）的所有权限

6、删除用户
delete from mysql.user where user='tester';
--删除用户

--7、删除数据库
drop database [schema名称|数据库名称];
--删除数据库

# 一、MySQL 创建用户

# 1、命令

CREATE USER 'username'@'host' IDENTIFIED BY 'password';
1

# 2、说明

  • username：你将创建的用户名
  • host：指定该用户在哪个主机上可以登陆，如果是本地用户可用localhost，如果想让该用户可以从任意远程主机登陆，可以使用通配符%
  • password：该用户的登陆密码，密码可以为空，如果为空则该用户可以不需要密码登陆服务器

# 3、示例

CREATE USER 'dog'@'localhost' IDENTIFIED BY '123456';
CREATE USER 'pig'@'192.168.1.101_' IDENDIFIED BY '123456';
CREATE USER 'pig'@'%' IDENTIFIED BY '123456';
CREATE USER 'pig'@'%' IDENTIFIED BY '';
CREATE USER 'pig'@'%';

# 二、MySQL 用户授权

# 1、命令

GRANT privileges ON databasename.tablename TO 'username'@'host'
1

# 2、说明

  • privileges：用户的操作权限，如SELECT，INSERT，UPDATE等，如果要授予所的权限则使用ALL
  • databasename：数据库名
  • tablename：表名，如果要授予该用户对所有数据库和表的相应操作权限则可用表示，如.*

# 3、示例

GRANT SELECT, INSERT ON test.user TO 'pig'@'%';
GRANT ALL ON *.* TO 'pig'@'%';
GRANT ALL ON maindataplus.* TO 'pig'@'%';
1
2
3

# 4、注意

  • 用以上命令授权的用户不能给其它用户授权，如果想让该用户可以授权，用以下命令:
GRANT privileges ON databasename.tablename TO 'username'@'host' WITH GRANT OPTION;
1

三、MySQL 用户设置/修改密码

# 1、命令

SET PASSWORD FOR 'username'@'host' = PASSWORD('newpassword');
1
  • 如果是当前登陆用户用:
SET PASSWORD = PASSWORD("newpassword");
1

# 2、示例

SET PASSWORD FOR 'pig'@'%' = PASSWORD("123456");
1

# 四、MySQL 撤销用户权限

# 1、命令

REVOKE privilege ON databasename.tablename FROM 'username'@'host';
1

# 2、说明

  • privilege, databasename, tablename：同授权部分

# 3、示例

REVOKE SELECT ON *.* FROM 'pig'@'%';
1

# 4、注意

  • 假如你在给用户'pig'@'%'授权的时候是这样的（或类似的）：
--假如你在给用户'pig'@'%'授权的时候是这样的（或类似的）
GRANT SELECT ON test.user TO 'pig'@'%';
--则在使用（如下语句）
REVOKE SELECT ON *.* FROM 'pig'@'%';
--并不能撤销该用户对test数据库中user表的SELECT 操作
